C Standard-Funktionen: Code-Beispiele

abs

Gibt den Absolutwert einer Ganzzahl zurück.


#include <stdlib.h>
#include <stdio.h>

int main() {
    int num = -123;
    int absolute_value = abs(num);
    printf("Der Absolutwert von %d ist %d\n", num, absolute_value); // Ausgabe: Der Absolutwert von -123 ist 123
    return 0;
}

#include <math.h>
#include <stdio.h>

    printf("acos(%.2f) = %f rad\n", x, result);
    return 0;
}

asctime

Konvertiert eine `struct tm`-Zeitstruktur in eine lesbare Zeichenkette.


#include <stdio.h>
#include <time.h>

int main () {
   time_t rawtime;
   struct tm * timeinfo;

   time(&rawtime);
   timeinfo = localtime(&rawtime);
   printf("Die aktuelle lokale Zeit ist: %s", asctime(timeinfo));

   return(0);
}

asctime_s

Eine sicherere Version von `asctime`, die Pufferüberläufe verhindert.


#define __STDC_WANT_LIB_EXT1__ 1
#include <stdio.h>
#include <time.h>

int main () {
   struct tm newtime;
   time_t long_time;
   char buf[26];

   time(&long_time);
   localtime_s(&newtime, &long_time);

   if (asctime_s(buf, sizeof(buf), &newtime) == 0) {
       printf("Die aktuelle lokale Zeit ist: %s", buf);
   } else {
       printf("Fehler bei asctime_s\n");
   }

   return(0);
}

clock

Gibt die seit dem Programmstart vergangene Prozessorzeit in Ticks zurück.


#include <time.h>
#include <stdio.h>

void long_running_function() {
    for(long i=0; i<100000000; i++);
}

int main () {
   clock_t start, end;
   double cpu_time_used;

   start = clock();
   long_running_function();
   end = clock();

   cpu_time_used = ((double) (end - start)) / CLOCKS_PER_SEC;
   printf("Die Funktion hat %f Sekunden Prozessorzeit benötigt.\n", cpu_time_used);

   return(0);
}

difftime

Berechnet die Differenz in Sekunden zwischen zwei `time_t`-Werten.


#include <stdio.h>

#include <unistd.h>

int main () {
   time_t start, end;
   double diff;

   time(&start);
   printf("Starte Timer...\n");
   sleep(2); // Warte 2 Sekunden
   time(&end);

   printf("Vergangene Zeit: %.2f Sekunden.\n", diff);

   return(0);
}

feof

Überprüft, ob das Dateiende-Kennzeichen für einen Stream gesetzt ist.


#include <stdio.h>

int main () {
   FILE *fp;
   int c;
   fp = fopen("file.txt","r");
   if(fp == NULL) {
      perror("Error in opening file");
      return(-1);
   }
   while(1) {
      c = fgetc(fp);
      if( feof(fp) ) { 
         break ;
      }
      printf("%c", c);
   }
   fclose(fp);
   return(0);
}

fflush

Leert den Ausgabepuffer eines Streams.


#include <stdio.h>
#include <unistd.h>

int main() {
   char buff[1024];
   memset( buff, '\0', sizeof( buff ));
   fprintf(stdout, "Going to set full buffering\n");
   setvbuf(stdout, buff, _IOFBF, 1024);
   fprintf(stdout, "This is tutorialspoint.com\n");
   fprintf(stdout, "This output will go into buff\n");
   fflush( stdout ); // Leert den Puffer und schreibt auf den Bildschirm
   fprintf(stdout, "and this will appear when the program exits\n");
   return(0);
}

fgetc

Liest das nächste Zeichen aus einem Stream und gibt es als `unsigned char` (in ein `int` umgewandelt) zurück.


#include <stdio.h>

int main () {
   FILE *fp;
   int c;
   fp = fopen("file.txt","r");
   if(fp == NULL) {
      perror("Error in opening file");
      return(-1);
   }
   printf("Lese Dateiinhalt:\n");
   do {
      c = fgetc(fp);
      if(c != EOF) {
         putchar(c);
      }
   } while(c != EOF);
   fclose(fp);
   return(0);
}

memmove

Kopiert `n` Bytes von `src` nach `dest`. Die Speicherbereiche dürfen sich überlappen.


#include <stdio.h>
#include <string.h>

int main () {
   char str[100] = "Dieser String wird verschoben.";
   
   // Verschiebe die ersten 12 Zeichen um 7 Positionen nach rechts
   memmove(str + 7, str, 12);
   printf("%s\n", str);
   
   return(0);
}

memset

Füllt die ersten `n` Bytes des Speicherbereichs, auf den `ptr` zeigt, mit dem konstanten Byte `value`.


#include <stdio.h>
#include <string.h>

int main () {
   char str[50];
   strcpy(str,"This is string.h library function");
   puts(str);
   
   // Fülle die ersten 7 Zeichen mit '$'
   memset(str, '$', 7);
   puts(str);
   
   return(0);
}

mktime

Konvertiert eine `struct tm`-Struktur in einen `time_t`-Wert.


#include <stdio.h>
#include <time.h>

int main () {
   time_t rawtime;
   struct tm * timeinfo;

   time ( &rawtime );
   timeinfo = localtime ( &rawtime );
   timeinfo->tm_year = 2000 - 1900; // Jahr 2000
   timeinfo->tm_mon = 0; // Januar
   timeinfo->tm_mday = 1; // 1. Tag

   mktime ( timeinfo );
   printf ("Der 1. Januar 2000 war ein: %s", asctime(timeinfo));

   return 0;
}

qsort

Sortiert ein Array.


#include <stdio.h>
#include <stdlib.h>

int compare (const void * a, const void * b) {
   return ( *(int*)a - *(int*)b );
}

int main () {
   int values[] = { 40, 10, 100, 90, 20, 25 };
   int n;
   qsort (values, 6, sizeof(int), compare);
   for (n=0; n<6; n++) {
      printf ("%d ", values[n]);
   }
   printf("\n");
   return(0);
}

strtok

Teilt einen String in eine Reihe von Tokens auf, basierend auf einem Trennzeichen.


#include <stdio.h>
#include <string.h>

int main () {
  char str[] = "Apfel,Banane,Kirsche";
  char *token = strtok(str, ",");
  
  while (token != NULL) {
    printf("%s\n", token);
    token = strtok(NULL, ",");
  }
  
  return(0);
}

strtok_s

Eine sicherere Version von `strtok`, die einen zusätzlichen Zeiger zur Speicherung des Kontexts verwendet.


#define __STDC_WANT_LIB_EXT1__ 1
#include <stdio.h>
#include <string.h>

int main () {
  char str[] = "Apfel,Banane,Kirsche";
  char * context = NULL;
  char * token = strtok_s(str, ",", &context);

  while (token != NULL) {
    printf("%s\n", token);
    token = strtok_s(NULL, ",", &context);
  }

  return(0);
}
